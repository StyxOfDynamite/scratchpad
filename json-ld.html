                    var xhr = new XMLHttpRequest();
                    xhr.open('GET', 'api.reviews.co.uk/json-ld/company/snippet?store=my-company');
                    xhr.send(null);
                    var container_element = document.getElementById('rich-snippets');
                    xhr.onreadystatechange = function () {
                        var complete = 4; // readyState 4 means the request is done.
                        var success = 200; // status 200 is a successful return.
                        if (xhr.readyState === complete) {
                            if (xhr.status === success) {
                                container_element.innerHTML = xhr.responseText;
                            }
                        } else {
                            console.log('Error: ' + xhr.status);
                            // An error occurred during the request.
                        }
                    }
